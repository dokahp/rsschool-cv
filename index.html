Code Example

Solution that strips all text that follows any of a set of comment markers passed in. Any whitespace at the end of the line should also be stripped out. 

    


    function stripComments(input, markers) {
        let arr = input.split('\n').map(el => {
        let strEl = String(el)
        let newElem = [strEl]
        markers.forEach(marker => {
            if (strEl.includes(marker)) {
                newElem = [...newElem, strEl.slice(0, strEl.indexOf(marker)).trim()]
            }
        })
        return newElem.sort((a, b) => a.length > b.length)[0]
        })
        return arr.join('\n')
    };
